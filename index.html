// --- クイズ終了時の処理（修正版） --- function finishQuestion(result) { clearInterval(timerInterval); isLocked = true; const data = currentEraData.questions[currentIdx]; document.getElementById('bg-image').style.animationPlayState = 'paused'; document.body.classList.remove('wrong-flash'); // 正解の文字列を取得（複数ある場合は最初の1つを表示） const correctAnswer = data.answers[0]; if (result === 'correct') { document.getElementById('dialogue').innerHTML = ` 【 正解 】
${data.name}： 見事だ！その通り、答えは「${correctAnswer}」であったな。 `; changePortrait('win'); } else { // 不正解・時間切れの場合 const msg = (result === 'timeout') ? "時は満ちた..." : "まだまだ修行が足りぬようだな。"; document.getElementById('dialogue').innerHTML = ` 【 ${result === 'timeout' ? '時間切れ' : '残念'} 】
${data.name}： ${msg}
正解は「${correctAnswer}」だ。 `; changePortrait('lose'); } // 解説エリアにも正解を明記 document.getElementById('info-area').innerHTML = `
正解：${correctAnswer}
解説： ${data.info} `; document.getElementById('info-area').style.display = 'block'; document.getElementById('user-answer').style.display = 'none'; document.getElementById('submit-btn').style.display = 'none'; document.getElementById('next-btn').style.display = 'block'; } // --- checkAttemptの修正（お手つき回数制限などを入れる場合もここ） --- function checkAttempt() { if (isLocked || timeLeft <= 0) return; const data = currentEraData.questions[currentIdx]; const ans = document.getElementById('user-answer').value.trim(); if (data.answers.includes(ans)) { correctCount++; const bonus = timeLeft * 50; totalScore += (1000 + bonus); finishQuestion('correct'); } else { // 間違えた時の演出 isLocked = true; timeLeft -= 5; // ペナルティを少し重く（3秒→5秒） updateTimerUI(); document.body.classList.add('wrong-flash'); changePortrait('lose'); if (timeLeft <= 0) { finishQuestion('timeout'); return; } setTimeout(() => { document.body.classList.remove('wrong-flash'); changePortrait('default'); document.getElementById('user-answer').value = ''; document.getElementById('user-answer').focus(); isLocked = false; }, 800); } }
